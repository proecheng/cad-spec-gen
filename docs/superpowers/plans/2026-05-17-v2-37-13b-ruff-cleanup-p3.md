### Step 2.4 — Write AC-8b consistency check script

- [ ] **S-2.4.1** Create `tmp/p3_e741_consistency_check.py`（不入仓）：
  ```python
  """AC-8b: validate rename target in decision table matches code.
  
  For each non-NOQA row in tmp/p3_e741_decisions.md, verify:
  - file <line> contains rename_target identifier
  - file <line> does NOT contain bare `\bl\b`
  
  Exit 0 if all 46 decisions satisfied; exit 1 otherwise.
  """
  import re
  import sys
  from pathlib import Path
  
  
  def main() -> int:
      decisions_md = Path("tmp/p3_e741_decisions.md").read_text(encoding="utf-8")
      failures: list[str] = []
      
      for line in decisions_md.splitlines():
          # Match `| file:line | 分类 | rename_target | 理由 |`
          m = re.match(r"\|\s*([^|]+\.py):(\d+)\s*\|\s*([^|]+?)\s*\|\s*([^|]+?)\s*\|", line)
          if not m:
              continue
          fpath, lineno, category, target = m.group(1).strip(), int(m.group(2)), m.group(3).strip(), m.group(4).strip()
          
          if target == "NOQA":
              # 数学保留 — verify noqa: E741 present on that line + 不 rename
              src_line = Path(fpath).read_text(encoding="utf-8").splitlines()[lineno - 1]
              if "noqa: E741" not in src_line:
                  failures.append(f"{fpath}:{lineno} NOQA decision but no `noqa: E741` in code")
          else:
              # Rename target should appear in file (allow ±2 line window for indent shifts)
              file_text = Path(fpath).read_text(encoding="utf-8").splitlines()
              window = "\n".join(file_text[max(0, lineno - 3):lineno + 2])
              if not re.search(rf"\b{re.escape(target)}\b", window):
                  failures.append(f"{fpath}:{lineno} target `{target}` not found in line window")
              if re.search(r"\bl\b", window) and target != "l":
                  failures.append(f"{fpath}:{lineno} bare `l` still in window (rename incomplete)")
      
      if failures:
          print(f"AC-8b FAIL ({len(failures)} issues):")
          for f in failures:
              print(f"  {f}")
          return 1
      print(f"AC-8b PASS — all 46 E741 decisions consistent with code")
      return 0
  
  
  if __name__ == "__main__":
      sys.exit(main())
  ```
