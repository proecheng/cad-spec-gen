# v2.24.0 Model Library Invocation Loop Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a v2.24.0 model-library invocation loop where user STEP selections, SolidWorks Toolbox candidates, codegen, and reports share one stable path and geometry-quality contract.

**Architecture:** Keep `PartsResolver` geometry decisions as the authoritative runtime record. Add small shared helpers for project paths and `parts_library.yaml` writes, then wire `cad_pipeline.py`, `tools/model_import.py`, `tools/model_audit.py`, and `codegen/gen_std_parts.py` through those helpers so data does not drift between review, import, codegen, and audit. Add a read-only SolidWorks export-plan artifact that lists executable Toolbox candidates without starting COM conversion.

**Tech Stack:** Python 3.10+, pytest, PyYAML, CadQuery STEP import/export, existing `cad_pipeline.py` CLI, existing `PartsResolver` and `SwToolboxAdapter`.

---

## Current Baseline

- `cad_spec_gen.py::_flatten_review_items()` preserves structured model-selection fields in `DESIGN_REVIEW.json`.

from __future__ import annotations

from dataclasses import dataclass
from pathlib import Path


def _resolve_root(project_root: str | Path) -> Path:
    return Path(project_root).expanduser().resolve()


def _subsystem_from_artifact(path: Path, project_root: Path) -> str | None:
    try:
        rel = path.resolve().parent.relative_to(project_root)
    except ValueError:
        return None
    parts = rel.parts
    if len(parts) >= 2 and parts[0] in {"output", "cad"}:
        return parts[1]
    return None


@dataclass(frozen=True)
class ModelProjectContext:
    project_root: Path
    subsystem: str | None = None

    def __post_init__(self) -> None:
        object.__setattr__(self, "project_root", _resolve_root(self.project_root))
        if self.subsystem == "":
            object.__setattr__(self, "subsystem", None)

    @classmethod
    def from_review_json(
        cls,
        review_json_path: str | Path,
        *,
        project_root: str | Path,
    ) -> "ModelProjectContext":
        root = _resolve_root(project_root)
        review = Path(review_json_path)
        return cls(
            project_root=root,
            subsystem=_subsystem_from_artifact(review, root),
        )

    @classmethod
    def for_subsystem(
        cls,
        subsystem: str | None,
        *,
        project_root: str | Path,
    ) -> "ModelProjectContext":
        return cls(project_root=project_root, subsystem=subsystem or None)

    @property
    def cad_dir(self) -> Path | None:
        if not self.subsystem:
            return None
        return self.project_root / "cad" / self.subsystem

    @property
    def meta_dir(self) -> Path:
        if self.cad_dir is None:
            return self.project_root / ".cad-spec-gen"
        return self.cad_dir / ".cad-spec-gen"

    @property
    def std_parts_dir(self) -> Path:
        return self.project_root / "std_parts"

    @property
    def user_provided_dir(self) -> Path:
        return self.std_parts_dir / "user_provided"

    @property
    def parts_library_path(self) -> Path:
        return self.project_root / "parts_library.yaml"

    @property
    def model_choices_path(self) -> Path:
        return self.meta_dir / "model_choices.json"

    @property
    def model_imports_path(self) -> Path:
        return self.meta_dir / "model_imports.json"

    @property
    def geometry_report_path(self) -> Path:
        return self.meta_dir / "geometry_report.json"

    @property
    def sw_export_plan_path(self) -> Path:
        return self.meta_dir / "sw_export_plan.json"
```

from __future__ import annotations

import os
from dataclasses import dataclass
from datetime import datetime, timezone
from pathlib import Path
from typing import Sequence

from tools.model_context import ModelProjectContext


@dataclass(frozen=True)
class UserStepMapping:
    part_no: str
    name_cn: str
    file_rel: str
    source_path: str
    source_hash: str
    bbox_mm: tuple[float, float, float] | None = None


def _utc_now() -> str:
    return datetime.now(timezone.utc).isoformat()


def _load_yaml(path: Path) -> dict:
    import yaml  # type: ignore

    if not path.is_file():
        return {"extends": "default", "mappings": []}
    with path.open(encoding="utf-8") as f:
        loaded = yaml.safe_load(f) or {}
    if not isinstance(loaded, dict):
        return {"extends": "default", "mappings": []}
    loaded.setdefault("extends", "default")
    if not isinstance(loaded.get("mappings"), list):
        loaded["mappings"] = []
    return loaded


def _dump_yaml_atomic(path: Path, payload: dict) -> None:
    import yaml  # type: ignore

    path.parent.mkdir(parents=True, exist_ok=True)
    tmp_path = path.with_suffix(path.suffix + ".tmp")
    with tmp_path.open("w", encoding="utf-8") as f:
        yaml.safe_dump(payload, f, allow_unicode=True, sort_keys=False)
    os.replace(tmp_path, path)


def _without_previous_user_mapping(
    mappings: Sequence[object],
    *,
    part_no: str,
) -> list[object]:
    kept: list[object] = []
    for mapping in mappings:
        if not isinstance(mapping, dict):
            kept.append(mapping)
            continue
        same_part = (mapping.get("match") or {}).get("part_no") == part_no
        user_owned = (mapping.get("provenance") or {}).get("provided_by_user")
        if same_part and user_owned:
            continue
        kept.append(mapping)
    return kept


def prepend_user_step_mapping(
    context: ModelProjectContext,
    mapping: UserStepMapping,
) -> Path:
    cfg = _load_yaml(context.parts_library_path)
    mappings = _without_previous_user_mapping(
        cfg.get("mappings") or [],
        part_no=mapping.part_no,
    )
    provenance = {
        "provided_by_user": True,
        "provided_at": _utc_now(),
        "source_path": mapping.source_path,
        "source_hash": mapping.source_hash,
        "name_cn": mapping.name_cn,
        "validated": True,
    }
    if mapping.bbox_mm is not None:
        provenance["bbox_mm"] = [float(v) for v in mapping.bbox_mm]
    cfg["mappings"] = [
        {
            "match": {"part_no": mapping.part_no},
            "adapter": "step_pool",
            "spec": {"file": mapping.file_rel},
            "provenance": provenance,
        },
        *mappings,
    ]
    _dump_yaml_atomic(context.parts_library_path, cfg)
    return context.parts_library_path
```

    bbox_mm: tuple[float, float, float] | None = None
    warnings: list[str] | None = None

    def to_dict(self) -> dict[str, Any]:
        return {
            "ok": self.ok,
            "reason": self.reason,
            "source_hash": self.source_hash,
            "bbox_mm": list(self.bbox_mm) if self.bbox_mm else None,
            "warnings": list(self.warnings or []),
        }


def _bbox_from_step(path: Path) -> tuple[float, float, float]:
    import cadquery as cq

    shape = cq.importers.importStep(str(path))
    bb = shape.val().BoundingBox()
    return (float(bb.xlen), float(bb.ylen), float(bb.zlen))


def validate_step_file(path: str | Path) -> StepValidationResult:
    step_path = Path(path)
    if step_path.suffix.lower() not in {".step", ".stp"}:
        return StepValidationResult(False, f"not a STEP file: {step_path}")
    try:
        head = step_path.read_text(encoding="utf-8", errors="ignore")[:4096]
    except OSError as exc:
        return StepValidationResult(False, f"cannot read STEP file: {exc}")
    if "ISO-10303-21" not in head:
        return StepValidationResult(False, f"invalid STEP header: {step_path}")
    source_hash = _sha256_file(step_path)
    try:
        bbox = _bbox_from_step(step_path)
    except Exception as exc:
        return StepValidationResult(False, f"invalid STEP geometry: {exc}", source_hash)
    if any(v <= 0 for v in bbox):
        return StepValidationResult(False, f"invalid STEP bbox: {bbox}", source_hash)
    warnings: list[str] = []
    if max(bbox) > 10000:
        warnings.append("bbox exceeds 10000 mm; units may be wrong")
    if min(bbox) < 0.01:
        warnings.append("bbox below 0.01 mm; model may be empty or units may be wrong")
    return StepValidationResult(True, source_hash=source_hash, bbox_mm=bbox, warnings=warnings)
```

from __future__ import annotations

import json
import os
from datetime import datetime, timezone
from pathlib import Path
from typing import Any

from adapters.parts.sw_toolbox_adapter import SwToolboxAdapter
from adapters.solidworks import sw_toolbox_catalog
from parts_resolver import PartQuery, PartsResolver
from tools.model_context import ModelProjectContext


def _utc_now() -> str:
    return datetime.now(timezone.utc).isoformat()


def _query_from_row(row: dict[str, Any], project_root: Path) -> PartQuery:
    return PartQuery(
        part_no=row.get("part_no", ""),
        name_cn=row.get("name_cn", ""),
        material=row.get("material", ""),
        category=row.get("category", ""),
        make_buy=row.get("make_buy", ""),
        project_root=str(project_root),
    )


def _cache_step_path(part, sw_cfg: dict[str, Any], config_name: str = "Default") -> Path:
    safe_config = "".join(c if c.isalnum() or c in "-_." else "_" for c in config_name)
    cache_root = sw_toolbox_catalog.get_toolbox_cache_root(sw_cfg)
    cache_stem = f"{Path(part.filename).stem}_{safe_config}"
    return cache_root / part.standard / part.subcategory / f"{cache_stem}.step"


def build_sw_export_plan(
    *,
    bom_rows: list[dict[str, Any]],
    registry: dict[str, Any],
    context: ModelProjectContext,
    adapter=None,
) -> dict[str, Any]:
    sw_cfg = registry.get("solidworks_toolbox", {})
    sw_adapter = adapter or SwToolboxAdapter(config=sw_cfg)
    ok, unavailable_reason = sw_adapter.is_available()
    resolver = PartsResolver(registry=registry, adapters=[])

    candidates: list[dict[str, Any]] = []
    for row in bom_rows:
        query = _query_from_row(row, context.project_root)
        for rule in resolver.matching_rules(query, adapter_name="sw_toolbox"):
            if not ok:
                candidates.append({
                    "part_no": query.part_no,
                    "name_cn": query.name_cn,
                    "adapter": "sw_toolbox",
                    "action": "unavailable",
                    "warnings": [unavailable_reason or "sw_toolbox unavailable"],
                })
                continue
            part = sw_adapter.find_sldprt(query, rule.get("spec", {}))
            if part is None:
                candidates.append({
                    "part_no": query.part_no,
                    "name_cn": query.name_cn,
                    "adapter": "sw_toolbox",
                    "action": "no_candidate",
                    "warnings": ["no matching Toolbox SLDPRT"],
                })
                continue
            step_path = _cache_step_path(part, sw_cfg)
            cache_state = "present" if step_path.is_file() else "missing"
            candidates.append({
                "part_no": query.part_no,
                "name_cn": query.name_cn,
                "adapter": "sw_toolbox",
                "standard": part.standard,
                "subcategory": part.subcategory,
                "sldprt_path": str(part.sldprt_path),
                "step_cache_path": str(step_path),
                "cache_state": cache_state,
                "config_match": "matched",
                "config_name": "Default",
                "action": "reuse_cache" if cache_state == "present" else "export",
                "warnings": [],
            })

    return {
        "schema_version": 1,
        "generated_at": _utc_now(),
        "project_root": str(context.project_root),
        "subsystem": context.subsystem,
        "candidates": candidates,
    }


def write_sw_export_plan(plan: dict[str, Any], context: ModelProjectContext) -> Path:
    context.meta_dir.mkdir(parents=True, exist_ok=True)
    out_path = context.sw_export_plan_path
    tmp_path = out_path.with_suffix(out_path.suffix + ".tmp")
    tmp_path.write_text(
        json.dumps(plan, ensure_ascii=False, indent=2),
        encoding="utf-8",
        newline="\n",
    )
    os.replace(tmp_path, out_path)
    return out_path
```
